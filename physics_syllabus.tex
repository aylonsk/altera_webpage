\section{Course Overview}
This course focuses on the mathematical foundations and physical applications of classical mechanics, designed to showcase the power of State-Aware Cognitive Architecture.

\section{Foundational Mathematics (Prerequisites)}
\begin{itemize}
    \item \textbf{Basic Differentiation}: Derivatives of polynomial functions.
    \item \textbf{Geometric Integration}: Calculating the area under a curve for linear and constant functions.
    \item \textbf{Vector Decomposition}: Resolving 2D vectors into orthogonal components.
\end{itemize}

\section{Module 1: Linear Kinematics}
\begin{itemize}
    \item \textbf{Position and Displacement}: Distinguishing between path length and relative position markers.
    \item \textbf{Instantaneous Velocity}: The formal definition as $\frac{dx}{dt}$. Requires mastery of \textit{Basic Differentiation}.
    \item \textbf{Constant Acceleration}: Derivation of kinematic equations from a constant $a$ value.
    \item \textbf{Graphical Analysis}: Relating $x(t)$, $v(t)$, and $a(t)$ graphs.
    \begin{itemize}
        \item \textit{Concept Dependency}: Finding displacement from $v(t)$ requires \textit{Geometric Integration}.
    \end{itemize}
\end{itemize}

\section{Module 2: Dynamics and Forces}
\begin{itemize}
    \item \textbf{Newton's Second Law}: The relationship $F_{net} = ma$.
    \item \textbf{Free Body Diagrams}: Systemic modeling of force vectors. Requires mastery of \textit{Vector Decomposition}.
    \item \textbf{Friction and Normal Force}: Modeling dissipative forces in mechanical systems.
\end{itemize}

\section{Module 3: Work and Energy}
\begin{itemize}
    \item \textbf{Work-Energy Theorem}: The integration of force over distance.
    \item \textbf{Conservation of Energy}: Modeling closed mechanical systems.
\end{itemize}
